SQL Queries: A Handy Cheat Sheet

Published on Nov 8, 2025 ¬∑ Tags: SQL

-- Count orders per customer
SELECT customer_id, COUNT(*) AS orders
FROM orders
GROUP BY customer_id
ORDER BY orders DESC;

-- Rolling 7-day average
SELECT
  date,
  value,
  AVG(value) OVER (ORDER BY date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) AS avg7
FROM metrics;
